\question \textbf{CORE M.12}
Determine whether each relation is an equivalence relation.
\begin{parts}
\part
Let $\sim$ be the relation on $\mathbb{Z}$ defined by $a\sim b$ if and only if $a+b$ is even.
\part
Let $\circleddash$ be the relation on $\mathbb{Z}$ defined by $a \circleddash b$ if and only if $ab\ge0$.
\part
Let $\barwedge$ be the relation on $\mathbb{Z}$ defined by $a \barwedge b$ if and only if $\gcd(a,b)=1$.
\part
Let $\bowtie$ be the relation on $\mathbb{Z}$ defined by $a \bowtie b$ if and only if $a$ has the same number of digits as $b$.
\end{parts}

\question \textbf{CORE M.13}
Each relation below is an equivalence relation. Describe the equivalence classes for each equivalence relation.
\begin{parts}
\part
In $\mathbb{R}\times \mathbb{R}$ $(a,b)\sim(x,y)$ if $\frac{y-b}{x-a}=2$.
\part
In $\mathbb{R}\times \mathbb{R}$ $(a,b)\sim(x,y)$ if $a^2+b^2=x^2+y^2$.
\part
In $\mathbb{Z}$ $a\sim b$ if $|a|=|b|$.
\part
In $\mathbb{Z}$ $a\sim b$ if $3a+4b=7n$ for some integer $n$.
\end{parts}

\question \textbf{M.14}
Let $n$ be a positive integer. Prove $[a]$ is a unit in $\mathbb{Z}_n$ if and only if $[a]x=[b]$ has a unique solution for each $[b]\in\mathbb{Z}_n$.

\question \textbf{CORE M.15}
\begin{parts}
\part Find all the units of $\mathbb{Z}_{18}$
\part Find all the zero divisors of $\mathbb{Z}_{18}$
\part Find the multiplicative inverse of $[5]$ in $\mathbb{Z}_{18}$.
\part Compute $[13]_{18}+[11]_{18}$ and $[13]_{18}\cdot[11]_{18}$ in $\mathbb{Z}_{18}$.
\end{parts}
